import math from PIL import Image, ImageDraw, ImageFont # フィギュアの仮画像を自動生成（再実行） def create_icon(label, size=64, color=(255, 255, 255), bg=(0, 100, 255)): img = Image.new("RGB", (size, size), bg) draw = ImageDraw.Draw(img) font = ImageFont.load_default() w, h = draw.textsize(label, font=font) draw.text(((size - w) / 2, (size - h) / 2), label, fill=color, font=font) return img # 30個分の仮アイコン生成 labels = [f"{i+1}" for i in range(30)] icons = [create_icon(label) for label in labels] # 画像を保存してファイルパス返す paths = [] for i, icon in enumerate(icons): path = f"/mnt/data/figure_icon_{i+1}.png" icon.save(path) paths.append(path) paths
